<script>
function showTopic(topicId) {
  // 1) 按钮样式
  document.querySelectorAll('.filter-link').forEach(el => el.classList.remove('active'));
  const btn = document.getElementById('filter-' + topicId);
  if (btn) btn.classList.add('active');

  // 2) 区块显示/隐藏
  document.querySelectorAll('.topic-section').forEach(sec => sec.classList.add('hidden'));
  const target = document.getElementById('section-' + topicId);
  if (target) target.classList.remove('hidden');
}
// 默认显示 de novo
document.addEventListener('DOMContentLoaded', () => showTopic('denovo'));
</script>
